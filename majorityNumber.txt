public class Solution {
    /**
     * @param nums: a list of integers
     * @return: find a  majority number
     */
    public int majorityNumber(ArrayList<Integer> nums) {
        // write your code
        //等下用Iterator遍历
        int[] key = new int[nums.size()];
        for(int i = 0;i < nums.size();i++){
            key[i] = nums.get(i);
        }
        Arrays.sort(key);
        return key[nums.size()/2];
    }
}
