class Solution {
public:
    /**
     * @param matrix: A list of lists of integers
     * @return: Void
     */
    void setZeroes(vector<vector<int> > &matrix) {
        // write your code here
        //“Ï≥£¥¶¿Ì
        int m = matrix.size();
        if(m == 0)  return;
        int n = matrix[0].size();
        bool row = false;
        bool col = false;
        for(int i = 0; i < m; ++i){
            if(matrix[i][0] == 0){
                row = true;
                break;
            }
        }
        for(int j = 0; j < n; ++j){
            if(matrix[0][j] == 0){
                col = true;
                break;
            }
        }
        for(int i = 1; i < m; ++i){
            for(int j =1; j < n; ++j){
                if(matrix[i][j] == 0){
                    matrix[i][0] = 0;
                    matrix[0][j] = 0;
                }
            }
        }
        for(int i = 1; i < m; ++i){
            for(int j =1; j < n; ++j){
                if(matrix[i][0] == 0 || matrix[0][j] == 0){
                    matrix[i][j] = 0;
                }
            }
        }
        if(row){
            for(int i = 0; i < m; ++i){
                matrix[i][0] = 0;
            }
        }
        if(col){
            for(int j = 0; j < n; ++j){
                matrix[0][j] = 0;
            }
        }
    }
};
